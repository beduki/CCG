                    if($_POST)
                    {
                        $num = $_POST['num1'];
                    $count = 0;
                    for ($i=1; $i <= $num; $i++) 
                    {
                        if (($num % $i) == 0)
                        {
                            $count++;
                        }
                    }

                        echo "$num is not a prime number.";
                    }
                }
